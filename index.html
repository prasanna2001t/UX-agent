async function runAgent() { const notes = document.getElementById('userInput').value; const outputDiv = document.getElementById('output'); const reviewDiv = document.getElementById('reviewNotes'); const btn = document.getElementById('genBtn'); if (!notes) return alert("Please enter notes!"); btn.disabled = true; btn.innerText = "Agent is working..."; try { const LAMBDA_URL = "https://yk2tf47ong3um7xxo2fgzhob7e0qaajx.lambda-url.ap-south-1.on.aws"; const response = await fetch(LAMBDA_URL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ notes: notes }) }); // Check if response is JSON or HTML Error const contentType = response.headers.get("content-type"); if (contentType && contentType.includes("application/json")) { const data = await response.json(); outputDiv.innerText = data.text; reviewDiv.innerText = data.reviewNotes; } else { const rawError = await response.text(); throw new Error(`Server returned non-JSON: ${rawError.substring(0, 100)}`); } } catch (err) { outputDiv.innerText = `Error: ${err.message}`; } finally { btn.disabled = false; btn.innerText = "Generate & Review"; } }
